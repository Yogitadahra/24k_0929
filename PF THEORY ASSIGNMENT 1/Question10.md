Problem#10

//num is a string

INPUT num

fixednum = ""

i = 0

while i < Length(num):

  IF num[i] == "9" AND num[i+1] == "0":
  
    fixednum = fixednum + num[i]
    
    i = i + 2
    
  ELSE
  
    fixednum = fixednum + num[i]
    
    i = i + 1
    
OUTPUT fixednum
